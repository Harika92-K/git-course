git config --global user.name "Harika"
//create a configuration details like username,email etc
git --version
//To check the version which is installed in our system
git init
//creates an empty local repo
git add <filename>
//adds the file to the staging area
git commit -m "commit message"
//commits the files to the local repo and maintains commit history
git tag
//list all the tags available
git push origin main
//pushes the changes to the remote repository
git pull origin main
//getting the changes from remote repository
git branch
//list all the available branches in a repo
git checkout -b feature
//creates a new branch and jump to feature
